1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "rabbitmq.fullname" . }}-service)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "rabbitmq.fullname" . }}-service'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "rabbitmq.fullname" . }}-service --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.management }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rabbitmq.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:{{ .Values.service.ports.management }} to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.service.ports.management }}:{{ .Values.service.ports.management }}
{{- end }}

2. Access RabbitMQ Management Console:
   kubectl port-forward service/{{ include "rabbitmq.fullname" . }}-service {{ .Values.service.ports.management }}:{{ .Values.service.ports.management }}

3. RabbitMQ Connection Details:
   Host: {{ include "rabbitmq.fullname" . }}-service
   Port: {{ .Values.service.ports.amqp }}
   Username: {{ .Values.rabbitmq.username }}
   Password: {{ .Values.rabbitmq.password }}

4. Management Console:
   URL: http://localhost:{{ .Values.service.ports.management }}
   Username: {{ .Values.rabbitmq.username }}
   Password: {{ .Values.rabbitmq.password }}
