1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "cockroachdb.fullname" . }}-public)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "cockroachdb.fullname" . }}-public'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "cockroachdb.fullname" . }}-public --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.http }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "cockroachdb.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Access CockroachDB Admin Console:
   kubectl port-forward service/{{ include "cockroachdb.fullname" . }}-public 8080:{{ .Values.service.ports.http }}

3. Connect to CockroachDB SQL:
   kubectl exec -it $(kubectl get pods -l "app.kubernetes.io/name={{ include "cockroachdb.name" . }}" -o jsonpath="{.items[0].metadata.name}") -- /cockroach/cockroach sql --insecure

4. Create a database:
   kubectl exec -it $(kubectl get pods -l "app.kubernetes.io/name={{ include "cockroachdb.name" . }}" -o jsonpath="{.items[0].metadata.name}") -- /cockroach/cockroach sql --insecure -e "CREATE DATABASE IF NOT EXISTS mydb;"
